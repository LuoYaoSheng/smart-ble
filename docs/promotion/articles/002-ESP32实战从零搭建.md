## 二、BLE 服务设计

### 2.1 服务结构

我们设计一个简单的灯控服务：

```
LED Service
├── LED Control Characteristic (Write)
│   ├── 0xFF01 - LED ON (常亮)
│   ├── 0xFF00 - LED OFF (关闭)
│   ├── 0xFF02 - Fast Blink (快闪)
│   └── 0xFF03 - Slow Blink (慢闪)
└── LED Status Characteristic (Notify)
    └── 实时状态通知
```

### 2.2 UUID 定义

```cpp
// LED 服务 UUID
#define LED_SERVICE_UUID        "4FAFC201-1FB5-459E-8FCC-C5C9C331914B"

// 控制特征值 UUID
#define LED_CONTROL_CHAR_UUID   "BEB5483E-36E1-4688-B7F5-EA07361B26A8"

// 状态特征值 UUID
#define LED_STATUS_CHAR_UUID    "BEB5483E-36E1-4688-B7F5-EA07361B26A9"
```

## 三、ESP32 代码实现

### 3.1 初始化 BLE

```cpp
#include <esp_bt.h>
#include <esp_bt_main.h>
#include <esp_gap_ble_api.h>
#include <esp_gatts_api.h>

#define TAG "ESP32_BLE"

// 全局变量
static uint16_t led_service_handle = 0;
static uint16_t led_control_char_handle = 0;
static uint16_t led_status_char_handle = 0;

// GAP 参数
static esp_ble_adv_params_t adv_params = {
    .adv_int_min = 0x20,    // 最小广播间隔 20ms
    .adv_int_max = 0x40,    // 最大广播间隔 40ms
    .adv_type = ADV_TYPE_IND,
    .own_addr_type = BLE_ADDR_TYPE_PUBLIC,
    .channel_map = ADV_CHNL_ALL,
    .adv_filter_policy = ADV_FILTER_ALLOW_SCAN_ANY_CON_ANY,
};

// 初始化 BLE 控制器
void ble_init() {
    ESP_ERROR_CHECK(esp_bt_controller_mem_release(ESP_BT_MODE_CLASSIC_BT));

    esp_bt_controller_config_t bt_cfg = BT_CONTROLLER_INIT_CONFIG_DEFAULT();
    ESP_ERROR_CHECK(esp_bt_controller_init(&bt_cfg));
    ESP_ERROR_CHECK(esp_bt_controller_enable(ESP_BT_MODE_BLE));

    ESP_ERROR_CHECK(esp_bluedroid_init());
    ESP_ERROR_CHECK(esp_bluedroid_enable());

    ESP_ERROR_CHECK(esp_ble_gatts_register_callback(gatts_event_handler));
    ESP_ERROR_CHECK(esp_ble_gatt_set_local_mtu(500));

    ESP_LOGI(TAG, "BLE 初始化完成");
}
```

### 3.2 创建服务

```cpp
// LED 控制特征值
static const uint8_t led_control_char_value[4] = {0};

// 特征值属性 - 可写
static const uint8_t led_control_char_prop = ESP_GATT_CHAR_PROP_BIT_WRITE;

// LED 状态特征值
static uint8_t led_status_char_value[2] = {0, 0};

// 特征值属性 - 可通知
static const uint8_t led_status_char_prop = ESP_GATT_CHAR_PROP_BIT_NOTIFY;

// 创建服务
void create_led_service() {
    // LED 服务
    esp_gatt_srvc_id_t service_id = {
        .is_primary = true,
        .id = {
            .uuid = {
                .len = ESP_UUID_LEN_128,
                .uuid = {LED_SERVICE_UUID},
            },
            .inst_id = 0,
        },
    };

    // 创建服务
    esp_ble_gatts_create_service(
        &service_id,
        GATTS_HANDLE_NUM,
        &gatts_if
    );
}
```

### 3.3 创建特征值

```cpp
void create_led_characteristics() {
    // 控制特征值
    esp_bt_uuid_t control_char_uuid = {
        .len = ESP_UUID_LEN_128,
        .uuid = {LED_CONTROL_CHAR_UUID},
    };

    esp_attr_desc_t control_attr_desc = {
        .uuid = &control_char_uuid,
        .perm = ESP_GATT_PERM_WRITE,
        .max_length = 4,
        .length = sizeof(led_control_char_value),
        .value = led_control_char_value,
    };

    // 添加控制特征值
    esp_ble_gatts_add_char(
        led_service_handle,
        &control_attr_desc,
        &led_control_char_prop
    );

    // 状态特征值
    esp_bt_uuid_t status_char_uuid = {
        .len = ESP_UUID_LEN_128,
        .uuid = {LED_STATUS_CHAR_UUID},
    };

    esp_attr_desc_t status_attr_desc = {
        .uuid = &status_char_uuid,
        .perm = ESP_GATT_PERM_READ,
        .max_length = 2,
        .length = sizeof(led_status_char_value),
        .value = led_status_char_value,
    };

    // 添加状态特征值
    esp_ble_gatts_add_char(
        led_service_handle,
        &status_attr_desc,
        &led_status_char_prop
    );
}
```

### 3.4 处理写入事件

```cpp
// LED 模式
enum LEDMode {
    LED_OFF = 0xFF00,
    LED_ON = 0xFF01,
    LED_FAST_BLINK = 0xFF02,
    LED_SLOW_BLINK = 0xFF03,
};

LEDMode current_led_mode = LED_OFF;

void process_led_command(const uint8_t *data, uint16_t len) {
    if (len < 2) return;

    uint16_t cmd = (data[0] << 8) | data[1];

    switch (cmd) {
        case LED_ON:
            ESP_LOGI(TAG, "LED 开启");
            gpio_set_level(LED_GPIO, 1);
            current_led_mode = LED_ON;
            break;

        case LED_OFF:
            ESP_LOGI(TAG, "LED 关闭");
            gpio_set_level(LED_GPIO, 0);
            current_led_mode = LED_OFF;
            break;

        case LED_FAST_BLINK:
            ESP_LOGI(TAG, "LED 快闪");
            current_led_mode = LED_FAST_BLINK;
            break;

        case LED_SLOW_BLINK:
            ESP_LOGI(TAG, "LED 慢闪");
            current_led_mode = LED_SLOW_BLINK;
            break;

        default:
            ESP_LOGW(TAG, "未知命令: 0x%04X", cmd);
            break;
    }

    // 通知状态变化
    notify_led_status();
}

void notify_led_status() {
    uint8_t status[2];
    status[0] = (current_led_mode >> 8) & 0xFF;
    status[1] = current_led_mode & 0xFF;

    esp_ble_gatts_send_indicate(
        gatts_if,
        conn_id,
        led_status_char_handle,
        sizeof(status),
        status,
        false
    );
}

// GATT 事件处理
void gatts_event_handler(
    esp_gatts_cb_event_t event,
    esp_gatt_if_t gatts_if,
    esp_ble_gatts_cb_param_t *param
) {
    switch (event) {
        case ESP_GATTS_REG_EVT:
            // 服务注册完成，创建特征值
            create_led_characteristics();
            break;

        case ESP_GATTS_CONNECT_EVT:
            ESP_LOGI(TAG, "设备已连接");
            conn_id = param->connect.conn_id;
            break;

        case ESP_GATTS_DISCONNECT_EVT:
            ESP_LOGI(TAG, "设备已断开");
            // 重新开始广播
            esp_ble_gap_start_advertising(&adv_params);
            break;

        case ESP_GATTS_WRITE_EVT:
            // 处理写入事件
            process_led_command(
                param->write.value,
                param->write.len
            );
            break;

        default:
            break;
    }
}
```

### 3.5 开始广播

```cpp
void start_advertising() {
    esp_ble_adv_data_t adv_data = {
        .set_scan_rsp = false,
        .include_name = true,
        .include_txpower = true,
        .min_interval = 0x20,
        .max_interval = 0x40,
        .appearance = 0x00,
        .manufacturer_len = 0,
        .p_manufacturer_data = NULL,
        .service_uuid_len = sizeof(led_service_uuid),
        .p_service_uuid = (uint8_t *)led_service_uuid,
        .service_data_len = 0,
        .p_service_data = NULL,
        .flag = (ESP_BLE_ADV_FLAG_GEN_DISC | ESP_BLE_ADV_FLAG_BREDR_NOT_SPT),
    };

    esp_ble_gap_config_adv_data_raw(&adv_data);
}

// 主函数
void app_main() {
    // 初始化 GPIO
    gpio_config_t io_conf = {
        .pin_bit_mask = (1ULL << LED_GPIO),
        .mode = GPIO_MODE_OUTPUT,
    };
    gpio_config(&io_conf);

    // 初始化 BLE
    ble_init();
    create_led_service();
    start_advertising();

    ESP_LOGI(TAG, "LED 控制服务已启动");
}
```

## 四、使用 Smart BLE App 控制

### 4.1 连接设备

1. 打开 Smart BLE App
2. 开始扫描，找到你的 ESP32 设备
3. 点击连接

### 4.2 写入控制命令

| 模式 | HEX 数据 | UTF-8 数据 |
|------|----------|-----------|
| 开灯 | FF 01 | (二进制) |
| 关灯 | FF 00 | (二进制) |
| 快闪 | FF 02 | (二进制) |
| 慢闪 | FF 03 | (二进制) |

在 Smart BLE App 中：
1. 找到 `LED Control Characteristic`
2. 选择 HEX 格式
3. 输入 `FF01`
4. 点击写入

### 4.3 订阅状态通知

1. 找到 `LED Status Characteristic`
2. 点击"启用通知"
3. 每次状态变化都会收到通知

### 5.1 JSON 协议设计

```json
{
  "cmd": "led",
  "action": "on",
  "params": {
    "mode": "fast"
  }
}
```

### 5.2 解析 JSON

```cpp
#include <cJSON.h>

void process_json_command(const char *json_str) {
    cJSON *root = cJSON_Parse(json_str);
    if (root == NULL) {
        ESP_LOGE(TAG, "JSON 解析失败");
        return;
    }

    cJSON *cmd = cJSON_GetObjectItem(root, "cmd");
    if (cJSON_IsString(cmd) && strcmp(cmd->valuestring, "led") == 0) {
        cJSON *action = cJSON_GetObjectItem(root, "action");

        if (cJSON_IsString(action)) {
            if (strcmp(action->valuestring, "on") == 0) {
                gpio_set_level(LED_GPIO, 1);
            } else if (strcmp(action->valuestring, "off") == 0) {
                gpio_set_level(LED_GPIO, 0);
            } else if (strcmp(action->valuestring, "blink") == 0) {
                cJSON *params = cJSON_GetObjectItem(root, "params");
                cJSON *mode = cJSON_GetObjectItem(params, "mode");
                // 处理闪烁模式
            }
        }
    }

    cJSON_Delete(root);
}
```
