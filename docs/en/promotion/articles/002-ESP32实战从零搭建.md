enum LEDMode {
    LED_OFF = 0xFF00,
    LED_ON = 0xFF01,
    LED_FAST_BLINK = 0xFF02,
    LED_SLOW_BLINK = 0xFF03,
};

LEDMode current_led_mode = LED_OFF;

void process_led_command(const uint8_t *data, uint16_t len) {
    if (len < 2) return;

    uint16_t cmd = (data[0] << 8) | data[1];

    switch (cmd) {
        case LED_ON:
            ESP_LOGI(TAG, "LED 寮€鍚?);
            gpio_set_level(LED_GPIO, 1);
            current_led_mode = LED_ON;
            break;

        case LED_OFF:
            ESP_LOGI(TAG, "LED 鍏抽棴");
            gpio_set_level(LED_GPIO, 0);
            current_led_mode = LED_OFF;
            break;

        case LED_FAST_BLINK:
            ESP_LOGI(TAG, "LED 蹇棯");
            current_led_mode = LED_FAST_BLINK;
            break;

        case LED_SLOW_BLINK:
            ESP_LOGI(TAG, "LED 鎱㈤棯");
            current_led_mode = LED_SLOW_BLINK;
            break;

        default:
            ESP_LOGW(TAG, "鏈煡鍛戒护: 0x%04X", cmd);
            break;
    }

    // 閫氱煡鐘舵€佸彉鍖?
    notify_led_status();
}

void notify_led_status() {
    uint8_t status[2];
    status[0] = (current_led_mode >> 8) & 0xFF;
    status[1] = current_led_mode & 0xFF;

    esp_ble_gatts_send_indicate(
        gatts_if,
        conn_id,
        led_status_char_handle,
        sizeof(status),
        status,
        false
    );
}

// GATT 浜嬩欢澶勭悊
